// Gibt alle geraden Zahlen in doppelter Ausführung zurück, wenn sie größer als 10 sind
function filterEven(nums) {
  return nums.flatMap(n => (n % 2 === 0 ? (n > 10 ? [n, n] : [n]) : []));
}
console.log(filterEven([2, 4, 11, 15, 7, 8, 12]));
